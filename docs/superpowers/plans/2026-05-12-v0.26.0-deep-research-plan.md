# lib/deep-research.sh
# lib/deep-research.sh — helpers for /clone-wars:deep-research.
# Sourced. Depends on lib/log.sh, lib/state.sh, lib/consult.sh, lib/commanders.sh.
#
# Public:
#   cw_deep_research_compute_per_branch_timeout <total-s> <rounds> <K>
#       — ceiling-divide total budget across all planned branches
#   cw_deep_research_extract_metric <topic-text>
#       — heuristic metric name extraction; empty string if ambiguous
#   cw_deep_research_validate_result_json <path>
#       — schema check; rc=0 valid; rc>0 invalid (stderr)
#   cw_deep_research_extract_approaches <landscape-md-path>
#       — TSV from meditate landscape ## Approaches section
#   cw_deep_research_allocate_commanders <round> <K>
#       — K codex-eligible commanders, mod-rotated per round
#   cw_deep_research_check_convergence <slug> <round>
#       — rc=0 if delta<1% vs prior round; rc=1 otherwise

# cw_deep_research_compute_per_branch_timeout <total-s> <rounds> <K>
# Ceiling-divide total budget across (rounds * K) branches.
# Errors with rc=2 if K or rounds is 0 or negative.
cw_deep_research_compute_per_branch_timeout() {
  local total="${1:-}" rounds="${2:-}" k="${3:-}"
  [[ "$total" =~ ^[0-9]+$ ]] || { echo "total must be positive integer" >&2; return 2; }
  [[ "$rounds" =~ ^[1-9][0-9]*$ ]] || { echo "rounds must be positive integer" >&2; return 2; }
  [[ "$k" =~ ^[1-9][0-9]*$ ]] || { echo "K must be positive integer" >&2; return 2; }
  local total_branches=$((rounds * k))
  local result=$(( (total + total_branches - 1) / total_branches ))
  printf '%d\n' "$result"
}

# cw_deep_research_extract_metric <topic-text>
# Heuristic metric extractor. Whole-word case-insensitive match against
# a canonical metric vocabulary. Returns first match (lexically by topic
# position) to stdout; empty string if no match.
cw_deep_research_extract_metric() {
  local topic="${1:-}"
  [[ -n "$topic" ]] || { echo ""; return 0; }
  local lower; lower=$(printf '%s' "$topic" | tr '[:upper:]' '[:lower:]')
  # Vocabulary — alphabetically ordered for determinism within a single token,
  # but iteration is by topic position (whichever appears first wins).
  local -a vocab=(
    accuracy auc cost f1 latency loss memory params precision recall throughput
  )
  local best_pos=999999 best_word=""
  local word
  for word in "${vocab[@]}"; do
    # Whole-word match
    if [[ " $lower " =~ [^a-z0-9]"$word"[^a-z0-9] ]]; then
      # Find position
      local pre="${lower%%$word*}"
      local pos=${#pre}
      if (( pos < best_pos )); then
        best_pos=$pos
        best_word=$word
      fi
    fi
  done
  printf '%s\n' "$best_word"
}

# cw_deep_research_validate_result_json <relative-path>
# Caller must have cd'd to the branch dir (so log_paths relative resolution
# matches what trooper wrote). Returns 0 on valid; >0 on invalid (stderr).
cw_deep_research_validate_result_json() {
  local path="${1:-}"
  [[ -f "$path" ]] || { echo "result.json not found: $path" >&2; return 1; }
  # Prefer jq if available — much more robust JSON parsing
  if command -v jq >/dev/null 2>&1; then
    _cw_deep_research_validate_result_jq "$path"
  else
    _cw_deep_research_validate_result_awk "$path"
  fi
}

_cw_deep_research_validate_result_jq() {
  local path="$1"
  # Step 1: parse must succeed
  jq empty "$path" 2>/dev/null || { echo "malformed JSON" >&2; return 1; }
  # Step 2: required fields present
  local f
  for f in branch_id approach_label metric_name metric_value status runtime_s log_paths; do
    if [[ "$(jq -r "has(\"$f\")" "$path")" != "true" ]]; then
      echo "missing required field: $f" >&2; return 1
    fi
  done
  # Step 3: status enum
  local status; status=$(jq -r '.status' "$path")
  case "$status" in
    ok|fail|timeout|cost_blown) ;;
    *) echo "invalid status: $status" >&2; return 1 ;;
  esac
  # Step 4: metric_value null iff status != ok
  local mv_null; mv_null=$(jq -r '.metric_value == null' "$path")
  if [[ "$status" == "ok" && "$mv_null" == "true" ]]; then
    echo "status=ok requires non-null metric_value" >&2; return 1
  fi
  if [[ "$status" != "ok" && "$mv_null" == "false" ]]; then
    echo "status=$status requires null metric_value" >&2; return 1
  fi
  # Step 5: log_paths exist
  local p
  while IFS= read -r p; do
    [[ -f "$p" ]] || { echo "log_path missing: $p" >&2; return 1; }
  done < <(jq -r '.log_paths[]' "$path")
  return 0
}

_cw_deep_research_validate_result_awk() {
  local path="$1"
  # Minimal regex-based fallback. Less robust than jq; v1 acceptable.
  # Rejects clearly malformed; passes clearly valid; gray area requires jq.
  local body; body=$(<"$path")
  echo "$body" | grep -q '"branch_id"' || { echo "missing branch_id (awk fallback)" >&2; return 1; }
  echo "$body" | grep -q '"approach_label"' || { echo "missing approach_label" >&2; return 1; }
  echo "$body" | grep -q '"metric_name"' || { echo "missing metric_name" >&2; return 1; }
  echo "$body" | grep -q '"status"' || { echo "missing status" >&2; return 1; }
  echo "$body" | grep -q '"runtime_s"' || { echo "missing runtime_s" >&2; return 1; }
  echo "$body" | grep -q '"log_paths"' || { echo "missing log_paths" >&2; return 1; }
  # Status enum
  if ! echo "$body" | grep -qE '"status"[[:space:]]*:[[:space:]]*"(ok|fail|timeout|cost_blown)"'; then
    echo "invalid status" >&2; return 1
  fi
  # Coarse metric_value/status consistency
  local is_ok=0 is_null=0
  echo "$body" | grep -q '"status"[[:space:]]*:[[:space:]]*"ok"' && is_ok=1
  echo "$body" | grep -q '"metric_value"[[:space:]]*:[[:space:]]*null' && is_null=1
  if (( is_ok == 1 && is_null == 1 )); then
    echo "status=ok requires non-null metric_value (awk)" >&2; return 1
  fi
  # log_paths existence: extract crude — array of strings
  local log_line
  log_line=$(echo "$body" | grep -oE '"log_paths"[[:space:]]*:[[:space:]]*\[[^]]*\]')
  echo "$log_line" | grep -oE '"\./[^"]+"' | tr -d '"' | while read -r p; do
    [[ -f "$p" ]] || { echo "log_path missing: $p" >&2; exit 1; }
  done || return 1
  return 0
}
```

# cw_deep_research_extract_approaches <landscape-md-path>
# Parses ## Approaches section from a meditate landscape doc.
# Format expected (loose):
#   ## Approaches
#   N. **<label>** — <brief>
#   N. **<label>** — <brief>
#   ...
# Output: TSV "label\tbrief\n" per line; empty if no Approaches section.
cw_deep_research_extract_approaches() {
  local path="${1:-}"
  [[ -f "$path" ]] || { echo "landscape doc not found: $path" >&2; return 1; }
  awk '
    /^##[[:space:]]+Approaches[[:space:]]*$/ { in_section=1; next }
    /^##[[:space:]]/ && in_section { in_section=0 }
    in_section && /^[0-9]+\.[[:space:]]+\*\*/ {
      # Parse: "1. **<label>** — <brief>"
      line = $0
      sub(/^[0-9]+\.[[:space:]]+\*\*/, "", line)
      idx = index(line, "**")
      if (idx == 0) next
      label = substr(line, 1, idx - 1)
      rest = substr(line, idx + 2)
      sub(/^[[:space:]]*[—-][[:space:]]*/, "", rest)
      printf "%s\t%s\n", label, rest
    }
  ' "$path"
}
```

# rex is the canonical codex commander; the rest are unreserved by provider
# in v0.25.1 lib/commanders.sh + lib/consult.sh.
_CW_DEEP_RESEARCH_CMDR_POOL=(

# cw_deep_research_allocate_commanders <round> <K>
# Returns K unique commanders from the codex-eligible pool, mod-rotated
# per round. Round 1 takes positions 0..K-1; round 2 takes K..2K-1; etc.
# Errors with rc=2 if K > pool size or (round * K) > pool size.
cw_deep_research_allocate_commanders() {
  local round="${1:-}" k="${2:-}"
  [[ "$round" =~ ^[1-9][0-9]*$ ]] || { echo "round must be positive integer" >&2; return 2; }
  [[ "$k" =~ ^[1-9][0-9]*$ ]] || { echo "K must be positive integer" >&2; return 2; }
  local pool_size=${#_CW_DEEP_RESEARCH_CMDR_POOL[@]}
  if (( k > pool_size )); then
    echo "K=$k exceeds codex-eligible pool size ($pool_size)" >&2; return 2
  fi

#

#                                  [--time-budget DURATION] [--cost-warning USD]

#!/usr/bin/env bash
# bin/deep-research-score.sh — parse all branch result.json files in a round
# and produce scoreboard.md. Yoda's select phase reads scoreboard.md to pick
# survivors.

set -uo pipefail
PLUGIN_ROOT="${CLAUDE_PLUGIN_ROOT:-$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)}"
source "$PLUGIN_ROOT/lib/log.sh"
source "$PLUGIN_ROOT/lib/state.sh"

ROUND_DIR="$ART_DIR/round-$ROUND"
BRANCHES_FILE="$ROUND_DIR/branches.txt"
[[ -f "$BRANCHES_FILE" ]] || { log_error "no branches.txt for round $ROUND"; exit 1; }

# Collect (commander, branch_id, label) tuples from branches.txt
declare -a ROWS=()  # each row: branch_id\tcommander\tmetric_value\tstatus\truntime_s\tlabel
while IFS=$'\t' read -r bid cmdr label brief; do
  bd="$ART_DIR/round-$ROUND-$cmdr-$bid"
  result="$bd/result.json"
  if [[ ! -f "$result" ]]; then
    ROWS+=("${cmdr}-${bid}\t${cmdr}\tnull\tmissing\t0\t${label}")
    continue
  fi
  if ! ( cd "$bd" && cw_deep_research_validate_result_json result.json 2>/dev/null ); then
    ROWS+=("${cmdr}-${bid}\t${cmdr}\tnull\tinvalid\t0\t${label}")
    continue
  fi
  if command -v jq >/dev/null 2>&1; then
    mv=$(jq -r '.metric_value // "null"' "$result")
    st=$(jq -r '.status' "$result")
    rt=$(jq -r '.runtime_s' "$result")
  else
    mv=$(grep -oE '"metric_value"[[:space:]]*:[[:space:]]*[^,}]+' "$result" | sed 's/.*://; s/[[:space:]]*//g')
    st=$(grep -oE '"status"[[:space:]]*:[[:space:]]*"[^"]+"' "$result" | sed 's/.*"\([^"]*\)"$/\1/')
    rt=$(grep -oE '"runtime_s"[[:space:]]*:[[:space:]]*[0-9]+' "$result" | sed 's/.*://; s/[[:space:]]*//g')
  fi
  ROWS+=("${cmdr}-${bid}\t${cmdr}\t${mv}\t${st}\t${rt}\t${label}")
done < "$BRANCHES_FILE"

# Sort: ok branches first by metric_value descending, then non-ok at bottom by branch_id
TMP=$(mktemp)
trap 'rm -f "$TMP"' EXIT

# Print ok rows sorted descending by metric
printf '%b\n' "${ROWS[@]}" | awk -F'\t' '$4=="ok"' \
  | sort -t$'\t' -k3,3rg > "$TMP"

# Print non-ok rows sorted by branch_id
printf '%b\n' "${ROWS[@]}" | awk -F'\t' '$4!="ok"' \
  | sort -t$'\t' -k1,1 >> "$TMP"

# Render scoreboard.md

{
  echo "# Round $ROUND scoreboard"
  echo ""
  echo "| Rank | Branch | Commander | Metric | Status | Runtime | Approach |"
  echo "|---|---|---|---|---|---|---|"
  rank=1
  while IFS=$'\t' read -r bid cmdr mv st rt label; do
    echo "| $rank | $bid | $cmdr | $mv | $st | ${rt}s | $label |"
    rank=$((rank + 1))
  done < "$TMP"
} > "$TMP_OUT"

# Atomic write
mv "$TMP_OUT" "$SCOREBOARD"
log_ok "[score] round $ROUND scoreboard at $SCOREBOARD"
